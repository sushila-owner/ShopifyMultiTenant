const generateAd = async (product) => {
  // 1. Generate copy
  const copy = await openai.chat.completions.create({
    model: 'gpt-4',
    messages: [{
      role: 'user',
      content: `Create ad copy for: ${product.title}`
    }]
  });
  
  // 2. Create image with text overlay
  const image = await sharp(product.image)
    .resize(1080, 1080)
    .composite([{ input: textOverlay }])
    .toBuffer();
  
  // 3. Upload to S3
  const url = await uploadToS3(image);
  
  // 4. Save
  return await AdsCreative.create({
    merchantId: product.merchantId,
    productId: product._id,
    url,
    copy
  });
};
